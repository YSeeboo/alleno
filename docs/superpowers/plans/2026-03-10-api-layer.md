# API Layer Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a complete FastAPI REST layer on top of the service layer, covering 7 resource domains with Pydantic schemas, error handling, and TestClient tests.

**Architecture:** Each domain gets a dedicated `schemas/` file (Pydantic v2 models) and `api/` router file. A shared `api/_errors.py` context manager converts `ValueError` → 400, `RuntimeError` → 500. Tests use FastAPI's `TestClient` with the existing `db` fixture via dependency override.

**Tech Stack:** FastAPI, Pydantic v2 (`ConfigDict`, `from_attributes=True`), `starlette.testclient.TestClient`, SQLite in-memory for tests.

```python
from contextlib import contextmanager
from fastapi import HTTPException


@contextmanager
def service_errors():
    """Convert service-layer exceptions to HTTP responses."""
    try:
        yield
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except RuntimeError as e:
        raise HTTPException(status_code=500, detail=str(e))
```

No test needed — this is tested implicitly by every API test that exercises error paths.

```python
from typing import List, Optional
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session

from api._errors import service_errors
from database import get_db
from schemas.part import PartCreate, PartUpdate, PartResponse
from services.part import create_part, get_part, list_parts, update_part, delete_part

router = APIRouter(prefix="/parts", tags=["parts"])


@router.post("/", response_model=PartResponse)
def api_create_part(data: PartCreate, db: Session = Depends(get_db)):
    with service_errors():
        return create_part(db, data.model_dump(exclude_none=True))


@router.get("/", response_model=List[PartResponse])
def api_list_parts(category: Optional[str] = None, db: Session = Depends(get_db)):
    return list_parts(db, category=category)


@router.get("/{part_id}", response_model=PartResponse)
def api_get_part(part_id: str, db: Session = Depends(get_db)):
    part = get_part(db, part_id)
    if part is None:
        raise HTTPException(status_code=404, detail=f"Part {part_id} not found")
    return part


@router.patch("/{part_id}", response_model=PartResponse)
def api_update_part(part_id: str, data: PartUpdate, db: Session = Depends(get_db)):
    with service_errors():
        return update_part(db, part_id, data.model_dump(exclude_none=True))


@router.delete("/{part_id}", status_code=204)
def api_delete_part(part_id: str, db: Session = Depends(get_db)):
    part = get_part(db, part_id)
    if part is None:
        raise HTTPException(status_code=404, detail=f"Part {part_id} not found")
    with service_errors():
        delete_part(db, part_id)
```
